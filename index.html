Paths
ntui.paths.root

//Returns "NtUI.js" root folder.
//Requires the use of original framework filename.


Create element
ntui.createElement(tag, {attributes}, innerHTML);

//Example usage
var myElement = ntui.createElement(
"div",
{
class:"myObject",
style:"background: #000000;"
},
"This is my DOM object with dynamic javascript {myVar + 'string'}"
);


Render
ntui.render(element, target);

//Example usage

//Indexing and rendering elements
ntui.render(myElement, document.body);

//Rendering/updating all indexed elements
ntui.render();


CSS
//Raw CSS manipulation
ntui. css('body{background:#000000;}.myClass{color:#FF0000;}');


/* CSS Classes manipulation (.class) */


//Creates a new CSS class with optional raw CSS properties.
ntui.css. newClass(className, css);

//Replace all CSS properties of specified class with new raw CSS properties.
ntui.css. setClass(className, css);

//Add new raw CSS properties to the specified class without replacing.
ntui.css. addToClass(className, css);

//Sets an individual CSS property to the specified class.
ntui.css. setToClass(className, property, css);

//Gets a property value (css) from the specified class.
ntui.css. getFromClass(className, property);

//Removes a property from the specified class.
ntui.css. removeFromClass(className, property);

//Creates a new CSS object with optional raw CSS properties.
ntui.css. newObject(elementID, css);


/* CSS Objects manipulation (#object) */


//Replace all CSS properties of specified object with new raw CSS properties.
ntui.css. setObject(elementID, css);

//Add new raw CSS properties to the specified object without replacing.
ntui.css. addToObject(elementID, css);

//Sets an individual CSS property to the specified object.
ntui.css. setToObject(elementID, property, css);

//Gets a property value (css) from the specified object.
ntui.css. getFromObject(elementID, property);

//Removes a property from the specified object.
ntui.css. removeFromObject(elementID, property);

Main menu
//Adds a menu item, options is the same as HTML attributes.
ntui.ui.mainmenu. addItem(label, {options});

//Example:
ntui.ui.mainmenu. addItem(
'myButton',
{
onclick:'myFn();',
href:'#',
state:'auto',
}
);

//Toggle the state of menu items.
ntui.ui.mainmenu. toggleItem(itemID);



Update
ntui.ui.update();

//Recalculate the positions of UI components, normally is not necessary.


Theme
//Load a theme for NtUI.
ntui.ui.theme(theme, path);

//Get theme properties (CSS variables).
ntui.ui.theme.getProperty('--color-primary');

//Set theme properties (CSS variables).
ntui.ui.theme.setProperty('--color-primary', '#000000');

//Create theme properties (CSS variables).
ntui.ui.theme.newProperty('--myColor', '#FFFFFF');

//You can access to colors from HTML class too (default theme), example:

class="ntui-ui-background-primary-dark"

class="ntui-ui-background-accent"
